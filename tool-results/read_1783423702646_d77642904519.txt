   553→// ===== LIVE MAP COMPONENT =====
   554→function LiveMap({ riderLoc, driverLoc, from, to, status }: {
   555→  riderLoc?: { lat: number; lng: number } | null;
   556→  driverLoc?: { lat: number; lng: number } | null;
   557→  from?: string;
   558→  to?: string;
   559→  status?: string;
   560→}) {
   561→  // Build Google Maps embed URL with markers
   562→  let mapUrl = "";
   563→  if (driverLoc && riderLoc) {
   564→    // Show both markers with route
   565→    const midLat = ((driverLoc.lat + riderLoc.lat) / 2).toFixed(6);
   566→    const midLng = ((driverLoc.lng + riderLoc.lng) / 2).toFixed(6);
   567→    mapUrl = `https://www.google.com/maps?q=${midLat},${midLng}&z=14&output=embed`;
   568→  } else if (riderLoc) {
   569→    mapUrl = `https://www.google.com/maps?q=${riderLoc.lat},${riderLoc.lng}&z=15&output=embed`;
   570→  } else if (driverLoc) {
   571→    mapUrl = `https://www.google.com/maps?q=${driverLoc.lat},${driverLoc.lng}&z=15&output=embed`;
   572→  } else if (from && to) {
   573→    mapUrl = `https://www.google.com/maps?q=${encodeURIComponent(from + " to " + to + " Saudi Arabia")}&output=embed`;
   574→  } else if (from) {
   575→    mapUrl = `https://www.google.com/maps?q=${encodeURIComponent(from + " Saudi Arabia")}&output=embed`;
   576→  } else {
   577→    mapUrl = `https://www.google.com/maps?q=Riyadh&output=embed`;
   578→  }
   579→
   580→  return (
   581→    <div className="relative w-full h-full">
   582→      <iframe src={mapUrl} className="w-full h-full" style={{ border: 0 }} loading="lazy" />
   583→      {/* Location badges overlay */}
   584→      <div className="absolute top-2 right-2 flex flex-col gap-1">
   585→        {riderLoc && <div className="bg-blue-600 text-white text-xs px-2 py-1 rounded-lg flex items-center gap-1">🔵 {status === "ongoing" ? "الوجهة" : "الراكب"}</div>}
   586→        {driverLoc && <div className="bg-red-600 text-white text-xs px-2 py-1 rounded-lg flex items-center gap-1">🔴 السائق 🚗</div>}
   587→      </div>
   588→    </div>
   589→  );
   590→}

  1361→  const [tab, setTab] = useState<"login" | "register">("login");
  1362→  const [loading, setLoading] = useState(false);
  1363→  const [login, setLogin] = useState({ identifier: "", password: "" });
  1364→  const [reg, setReg] = useState({ name: "", email: "", phone: "", password: "", city: "الرياض", region: "الرياض" });
  1365→  const [showPwd, setShowPwd] = useState(false);
  1366→  const { toast } = useToast();
  1367→
  1368→  const handleLogin = async () => {
  1369→    if (!login.identifier || !login.password) { toast({ title: t("auth.enterCredentials", lang), variant: "destructive" }); return; }
  1370→    setLoading(true);
  1371→    try {
  1372→      const res = await fetch("/api/auth/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(login) });
  1373→      const data = await res.json();
  1374→      if (!res.ok) throw new Error(data.error);
  1375→      if (data.isBlocked) { toast({ title: t("auth.blocked", lang), description: data.blockReason, variant: "destructive" }); return; }
  1376→      onSuccess(data); setLogin({ identifier: "", password: "" });
  1377→    } catch (e) { toast({ title: t("auth.loginFailed", lang), description: e instanceof Error ? e.message : "", variant: "destructive" }); }
  1378→    finally { setLoading(false); }
  1379→  };
  1380→
  1381→  const handleRegister = async () => {
  1382→    if (!reg.name || !reg.email || !reg.phone || !reg.password) { toast({ title: t("auth.allFieldsRequired", lang), variant: "destructive" }); return; }
  1383→    setLoading(true);
  1384→    try {
  1385→      const res = await fetch("/api/auth/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(reg) });
  1386→      const data = await res.json();
  1387→      if (!res.ok) throw new Error(data.error);
  1388→      onSuccess(data); setReg({ name: "", email: "", phone: "", password: "", city: "الرياض", region: "الرياض" });
  1389→    } catch (e) { toast({ title: t("auth.registerFailed", lang), description: e instanceof Error ? e.message : "", variant: "destructive" }); }
  1390→    finally { setLoading(false); }
  1391→  };
  1392→
  1393→  const handleGoogleLogin = async () => {
  1394→    const email = prompt(lang === "ar" ? "أدخل بريد Google:" : "Enter Google email:", "user@gmail.com");
  1395→    if (!email) return;
  1396→    setLoading(true);
  1397→    try {
  1398→      const res = await fetch("/api/auth/google", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ googleId: `google_${Date.now()}`, email, name: email.split("@")[0], picture: null }) });
  1399→      const data = await res.json();
  1400→      if (!res.ok) throw new Error(data.error);
  1401→      onSuccess(data);
  1402→    } catch (e) { toast({ title: t("auth.loginFailed", lang), description: e instanceof Error ? e.message : "", variant: "destructive" }); }
  1403→    finally { setLoading(false); }
  1404→  };
  1405→
  1406→  return (
  1407→    <Dialog open={open} onOpenChange={onOpenChange}>
  1408→      <DialogContent className="sm:max-w-md">
  1409→        <DialogHeader><DialogTitle className="text-xl font-bold">{t("auth.welcome", lang)}</DialogTitle><DialogDescription>{t("auth.welcomeDesc", lang)}</DialogDescription></DialogHeader>
  1410→        <button onClick={handleGoogleLogin} className="w-full flex items-center justify-center gap-2 border border-zinc-300 rounded-lg h-12 hover:bg-zinc-50 transition-colors">
  1411→          <svg className="w-5 h-5" viewBox="0 0 24 24"><path fill="#4285F4" d="M22.56 12.25c0-.78-.07-1.53-.2-2.25H12v4.26h5.92c-.26 1.37-1.04 2.53-2.21 3.31v2.77h3.57c2.08-1.92 3.28-4.74 3.28-8.09z"/><path fill="#34A853" d="M12 23c2.97 0 5.46-.98 7.28-2.66l-3.57-2.77c-.98.66-2.23 1.06-3.71 1.06-2.86 0-5.29-1.93-6.16-4.53H2.18v2.84C3.99 20.53 7.7 23 12 23z"/><path fill="#FBBC05" d="M5.84 14.09c-.22-.66-.35-1.36-.35-2.09s.13-1.43.35-2.09V7.07H2.18C1.43 8.55 1 10.22 1 12s.43 3.45 1.18 4.93l2.85-2.22.81-.62z"/><path fill="#EA4335" d="M12 5.38c1.62 0 3.06.56 4.21 1.64l3.15-3.15C17.45 2.09 14.97 1 12 1 7.7 1 3.99 3.47 2.18 7.07l3.66 2.84c.87-2.6 3.3-4.53 6.16-4.53z"/></svg>
  1412→          <span className="font-medium text-black">{t("auth.loginGoogle", lang)}</span>
  1413→        </button>
  1414→        <div className="flex items-center gap-3"><div className="flex-1 h-px bg-zinc-200"></div><span className="text-xs text-zinc-400">{t("auth.or", lang)}</span><div className="flex-1 h-px bg-zinc-200"></div></div>
  1415→        <Tabs value={tab} onValueChange={(v) => setTab(v as "login" | "register")}>
  1416→          <TabsList className="grid grid-cols-2 w-full"><TabsTrigger value="login">{t("auth.login", lang)}</TabsTrigger><TabsTrigger value="register">{t("auth.register", lang)}</TabsTrigger></TabsList>

  1418→            <div><Label>{t("auth.emailOrPhone", lang)}</Label><Input value={login.identifier} onChange={(e) => setLogin({ ...login, identifier: e.target.value })} placeholder="grouthhacker@gmail.com" onKeyDown={(e) => { if (e.key === "Enter") handleLogin(); }} /></div>

  1421→            <div className="bg-zinc-50 rounded-lg p-3 text-xs text-zinc-600 text-center"><p className="font-bold mb-1">{t("auth.demoAccounts", lang)}</p><p>grouthhacker@gmail.com / Admin@2026</p><p>saad@example.com / 123456</p><p>ahmed@driver.com / 123456</p></div>
  1422→          </TabsContent>
  1423→          <TabsContent value="register" className="space-y-3 mt-4">
  1424→            <div><Label>{t("auth.name", lang)}</Label><Input value={reg.name} onChange={(e) => setReg({ ...reg, name: e.target.value })} /></div>

  1427→            <div><Label>{t("auth.password", lang)}</Label><Input type="password" value={reg.password} onChange={(e) => setReg({ ...reg, password: e.target.value })} /></div>
